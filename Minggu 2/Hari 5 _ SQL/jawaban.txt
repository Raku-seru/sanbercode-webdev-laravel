> Soal 2 Membuat Table di Dalam Database

CREATE TABLE users(
id int NOT NULL AUTO_INCREMENT PRIMARY KEY,
name varchar(255),
email varchar(255),
password varchar(255));
DESCRIBE users;

CREATE TABLE items(
id int NOT NULL AUTO_INCREMENT PRIMARY KEY,
name varchar(255),
description varchar(255),
price int,
stock int,
category_id int);
DESCRIBE items;

CREATE TABLE categories(
id int NOT NULL AUTO_INCREMENT PRIMARY KEY,
name varchar(255));
DESCRIBE categories;

ALTER TABLE items ADD CONSTRAINT category_id FOREIGN KEY(category_id) REFERENCES categories(id);

> Soal 4 Mengambil Data dari Database
a. Mengambil data users
SELECT id,name,email FROM users;

b. Mengambil data items
SELECT * FROM items WHERE price>1000000;
SELECT * FROM items WHERE name LIKE "%watch%";
SELECT * FROM items WHERE name LIKE "%sang%";
SELECT * FROM items WHERE name LIKE "%uniklo%";

c. Menampilkan data items join dengan kategori
SELECT * FROM items INNER JOIN categories ON items.category_id = categories.id; //cara 1
SELECT * FROM items,categories WHERE items.category_id = categories.id; //cara 2
